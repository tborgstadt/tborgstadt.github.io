# Convert passed categorical features to dummies, insert new dummy feature(s) next to original
def add_dummies(dfL, use_col):
    cols_to_create = np.unique(dfL[use_col])        # array of unique values in the passed col (use_col) of dfL
    for val in cols_to_create:
        new_col = use_col + "_" + val                              # append the value to the original field name
        tmp = np.zeros(len(dfL))                                   # initialze array same length as dfL
        tmp[dfL.loc[dfL[use_col] == val].index] = 1                # set to "1" where there is match in use_col
        dfL.insert(dfL.columns.get_loc(use_col) + 1, new_col, tmp) # insert new series like (numpy) into dfL
    return(dfL)

def clean_column_names(dfL):

Model Fit and Predict Function¶

This function is utilized for all the sci-kit learn classifiers. It accepts a classifier as primary input parameter. If grid search parameters are passed, it will perform a grid search and then fit the model based on the best estimator.

Since the dataset is imbalanced and the minority class is the class of interest (people making > 50K, or true '1'), the grid search will score based on 'recall'.

A 7-fold cross validation will be calculated using both 'recall' and 'accuracy'(or the default classifier scoring).

In [222]:
def Fit_Predict_Classifier(modelL, gridParmsL={}):     
    global X_train, X_test, y_train, y_test
    global df_feature_importance
    global fprL, tprL

    scoring_parm = 'recall'   # since we have imbalanced dataset and we are interested in minority class
    best = {}
    score = 0
        
    start = time.time()                 # start time
    
    if gridParmsL:
        grid = GridSearchCV(modelL, gridParmsL, scoring=scoring_parm, cv=7, n_jobs=-1)
        # http://scikit-learn.org/stable/modules/model_evaluation.html#scoring-parameter
        grid.fit(X_train, y_train)
        best = grid.best_params_
        score = round(grid.best_score_,3)
        if log: 
            #print "SCORES", grid.grid_scores_
            print 'Score based on: ' + scoring_parm
            print 'Best Score: ', grid.best_score_
            print 'Best Parameters: ', grid.best_params_
        # use the best estimator to fit and predict
        modelL = grid.best_estimator_

    # fit the model
    modelL.fit(X_train, y_train)
   
    # predict
    predicted = modelL.predict(X_test)

    prob = modelL.predict_proba(X_test)
    fprL, tprL, _ = roc_curve(y_test, modelL.predict_proba(X_test)[:,1]) 

    actual = y_test
    cm = confusion_matrix(actual, predicted)
    a = round(accuracy_score(actual, predicted),3)
    p = round(precision_score(actual, predicted),3)
    r = round(recall_score(actual, predicted),3)
    f1 = round(f1_score(actual, predicted),3)
    roc = round(roc_auc_score(actual, prob[:, 1]),3)
   
    
    # cross validation based on estimator
    cv = cross_val_score(modelL, X_train, y_train, scoring=None, cv=7, n_jobs=-1)
    cvm = round(cv.mean(),3)
    
    # cross validation based on the scoring_parm set above
    cvsp = cross_val_score(modelL, X_train, y_train, scoring=scoring_parm, cv=7, n_jobs=-1)
    cvspm = round(cvsp.mean(),3)

    try:
        df_feature_importance = modelL.get_feature_importances().sort_values(['effect'], ascending=False)
    except:
        df_feature_importance = []
    
    end = time.time()                   # end time
    elapsed = int(end - start)          # elapsed time   
    
    if log:
        modelL_desc = str(modelL).split('(')[0]
        print '\n----- ' + modelL_desc + ' -------'
        print'\nModel:'
        print modelL    
        print '\nCV Scoring Parm (' + scoring_parm + '):'
        print cvsp, ' Mean: ', cvspm
        print '\nCV Classifier Scoring Parm: '
        print cv, ' Mean: ', cvm
        print '\nOverall Accuracy:', a
        print
        print classification_report(actual, predicted)
        print confusion_matrix(actual, predicted, labels=None)
        print '\nROC AUC: ', roc
        
    est = str(modelL).replace('\n','')
    dte = str(time.strftime("%y%m%d.%H%M%S"))
    
    return([modelL_desc,gridParmsL,best,score,est,cm,roc,a,cvm,cvspm,p,r,f1,X_train.shape,elapsed,dte])
