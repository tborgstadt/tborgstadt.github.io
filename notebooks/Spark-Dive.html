def add_dummies(dfL, use_col):
    cols_to_create = np.unique(dfL[use_col])        # array of unique values in the passed col (use_col) of dfL
    for val in cols_to_create:
        new_col = use_col + "_" + val                              # append the value to the original field name
        tmp = np.zeros(len(dfL))                                   # initialze array same length as dfL
        tmp[dfL.loc[dfL[use_col] == val].index] = 1                # set to "1" where there is match in use_col
        dfL.insert(dfL.columns.get_loc(use_col) + 1, new_col, tmp) # insert new series like (numpy) into dfL
    return(dfL)

    global X_train, X_test, y_train, y_test
    global fprL, tprL

    start = time.time()                            # start time
    
    # fit the model
    modelL.fit(X_train, y_train)

    # predict
    predicted = modelL.predict(X_test)

    modelL_desc = str(modelL).split('(')[0]        # parse basic model name for reporting only
    
    # collect measures
    actual = y_test
    cm = confusion_matrix(actual, predicted)
    a = round(accuracy_score(actual, predicted),3)
    p = round(precision_score(actual, predicted),3)
    r = round(recall_score(actual, predicted),3)
    f1 = round(f1_score(actual, predicted),3)

    # probabilities and ROC
    try:                                           # error handling as not all classifiers have proba
        prob = modelL.predict_proba(X_test)
        fprL, tprL, _ = roc_curve(y_test, modelL.predict_proba(X_test)[:,1])
        roc = round(roc_auc_score(actual, prob[:, 1]),3)
        if note > '':
            legendName = str(modelL_desc) + str(note)
        else:
            legendName = modelL_desc
        Roc_Plot_Add(legendName)
    except:
        print '\nInelgible for ROC'
        roc = 0.000 
   
    # cross validation based on estimator
    cv = cross_val_score(modelL, X_train, y_train, scoring=None, cv=7, n_jobs=-1)
    cvm = round(cv.mean(),3)
    
    # cross validation based on the scoring_parm set above
    cvsp = cross_val_score(modelL, X_train, y_train, scoring='recall', cv=7, n_jobs=-1)
    cvspm = round(cvsp.mean(),3)
    
    print '\n-------------- Fit and Predict --------------------'
    print '\nModel:'
    print modelL
    print
    print classification_report(actual, predicted)

    if log:
        print
        print confusion_matrix(actual, predicted, labels=None)
        print '\nCV Recall Scoring Parm:'
        print cvsp, ' Mean: ', cvspm
        print '\nCV Classifier Scoring Parm: '
        print cv, ' Mean: ', cvm
        print '\nOverall Accuracy:', a
        print '\nROC AUC: ', roc
        
    est = str(modelL).replace('\n','')
    dte = str(time.strftime("%y%m%d.%H:%M:%S"))   # date time stamp
    end = time.time()                             # end time
    elapsed = int(end - start)                    # elapsed time for fit and predict
    
    Class_Results_Add([modelL_desc,est,cm,roc,a,cvm,cvspm,p,r,f1,X_train.shape,note,Env,elapsed,dte])
    return
